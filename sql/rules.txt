-- Alle Raum-Mitglieder können lesen
CREATE POLICY "rules_select_member" ON public.rules FOR SELECT
  USING (EXISTS (
    SELECT 1 FROM public.room_members rm
    WHERE rm.room_id = rules.room_id AND rm.user_id = auth.uid()
  ));

-- Nur Owner dürfen anlegen, ändern, löschen
CREATE POLICY "rules_insert_owner" ON public.rules FOR INSERT
  WITH CHECK (EXISTS (
    SELECT 1 FROM public.room_members rm
    WHERE rm.room_id = rules.room_id AND rm.user_id = auth.uid() AND rm.role = 'owner'
  ));

CREATE POLICY "rules_update_owner" ON public.rules FOR UPDATE
  USING (EXISTS (
    SELECT 1 FROM public.room_members rm
    WHERE rm.room_id = rules.room_id AND rm.user_id = auth.uid() AND rm.role = 'owner'
  ));

CREATE POLICY "rules_delete_owner" ON public.rules FOR DELETE
  USING (EXISTS (
    SELECT 1 FROM public.room_members rm
    WHERE rm.room_id = rules.room_id AND rm.user_id = auth.uid() AND rm.role = 'owner'
  ));
