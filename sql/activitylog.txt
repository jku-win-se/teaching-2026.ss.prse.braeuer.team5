CREATE TABLE activity_logs (
  id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  
  device_id UUID REFERENCES devices(id) ON DELETE SET NULL,
  room_id UUID REFERENCES rooms(id) ON DELETE SET NULL,
  
  user_id UUID REFERENCES auth.users(id) ON DELETE SET NULL,
  actor_type TEXT NOT NULL CHECK (actor_type IN ('user', 'automation', 'system')),
  
  action TEXT NOT NULL,
  new_value TEXT NOT NULL
);

-- 2. Performance-Indizes
CREATE INDEX idx_logs_created_at ON activity_logs(created_at DESC);
CREATE INDEX idx_logs_device_id ON activity_logs(device_id) WHERE device_id IS NOT NULL;
CREATE INDEX idx_logs_room_id ON activity_logs(room_id) WHERE room_id IS NOT NULL;
CREATE INDEX idx_logs_user_id ON activity_logs(user_id) WHERE user_id IS NOT NULL;

-- 3. Row Level Security (RLS) 
ALTER TABLE activity_logs ENABLE ROW LEVEL SECURITY;

-- 4. Policies
CREATE POLICY "Users can view all logs" 
ON activity_logs FOR SELECT 
TO authenticated 
USING (true);

CREATE POLICY "Users can insert logs" 
ON activity_logs FOR INSERT 
TO authenticated 
WITH CHECK (auth.uid() = user_id OR user_id IS NULL);
